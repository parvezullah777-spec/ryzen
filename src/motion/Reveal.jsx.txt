import React, { useEffect, useRef, useState } from "react";

/**
 * Wraps any existing section/element to animate it in on scroll.
 * Does NOT change the children's markup, classes, or logic —
 * just adds a wrapper div with reveal styling.
 *
 * Usage:
 *   <Reveal><YourExistingHeroSection /></Reveal>
 *   <Reveal delay={150} direction="left"><ProductCard {...p} /></Reveal>
 */
export default function Reveal({ children, delay = 0, direction = "up", as: Tag = "div", className = "" }) {
  const ref = useRef(null);
  const [visible, setVisible] = useState(false);

  useEffect(() => {
    const node = ref.current;
    if (!node) return;
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setTimeout(() => setVisible(true), delay);
          observer.unobserve(node);
        }
      },
      { threshold: 0.15, rootMargin: "0px 0px -60px 0px" }
    );
    observer.observe(node);
    return () => observer.disconnect();
  }, [delay]);

  const directionClass = direction === "left" ? "ryzen-reveal-left" : direction === "right" ? "ryzen-reveal-right" : "";

  return (
    <Tag
      ref={ref}
      className={`ryzen-reveal ryzen-anim-layer ${directionClass} ${visible ? "is-visible" : ""} ${className}`}
    >
      {children}
    </Tag>
  );
}
